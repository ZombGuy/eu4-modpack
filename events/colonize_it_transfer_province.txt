country_event = { #インドネシア
	id = colonize_it_transfer.1
	title = none
	desc = trade_company_region_indonesia
	picture = none
	hidden = yes
	trigger = {
		overlord = {
			any_owned_province = {
				trade_company_region = trade_company_indonesia
			}
		}
		is_colonial_nation = yes
		has_country_flag = colonize_it_trade_company_region
		has_country_flag = colonize_it_tag_indonesia
	}

	immediate = {
		overlord = {
			every_owned_province = {
				limit = {
					trade_company_region = trade_company_indonesia
				}
				cede_province = ROOT
				if = {
					limit = {
						is_core = overlord
					}
					add_core = ROOT
					remove_core = overlord
				}
			}
		}
	}
	option = {}
}
