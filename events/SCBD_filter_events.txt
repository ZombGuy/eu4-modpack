	}
	option = { #State
		name = SBCD_filter_events.state
		ai_chance = { factor = 0 }
		set_country_flag = SBCD_target_is_state_flag
	}
	option = { #Territory
		name = SBCD_filter_events.territory
		trigger = {
            event_target:SBCD_selected_country_target = {
                any_owned_province = {
                    SBCD_potential_province_triggers = yes
                    is_state = no
                }
            }
		}
		ai_chance = { factor = 0 }
		set_country_flag = SBCD_target_is_territory_flag
	}
	option = { #Trade Company
		name = SBCD_filter_events.trade_company
		trigger = {
            event_target:SBCD_selected_country_target = {
                any_owned_province = {

                    is_owned_by_trade_company = yes
                }
            }
		}
		ai_chance = { factor = 0 }
		set_country_flag = SBCD_target_is_trade_company_flag

	option = { #Clear
