ME_add_caravan_modifier = {
	add_country_modifier = {
		name = me_tim_the_caravans_$level$
		duration = -1
	}
}

ME_tim_add_spoils_of_war = {
    [[add]
	change_variable = {
		which = METimSpoilsOfWar
		value = $add$
	}
    ]
    [[remove]
	subtract_variable = {
		which = METimSpoilsOfWar
		value = $remove$
	}
    ]
    hidden_effect = {
        if = {
            limit = {
                check_variable = { 
                    which = METimSpoilsOfWar
                    value = 101
                }
            }
            set_variable = {
                which = METimSpoilsOfWar
                value = 100
            }
        }
        if = {
            limit = {
                NOT = {
                	check_variable = { 
                        which = METimSpoilsOfWar
                        value = 1
                    }
                }
                set_variable = {
                    which = METimSpoilsOfWar
                     value = 1
                }
            }
        }
    }
	hidden_effect = {
		ME_TIM_remove_spoils_of_war = yes
		ME_tim_add_spoils_of_war_modifier = yes
	}
}
